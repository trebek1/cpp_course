Can dereference pointers to compare values 

string s1 {"Frank"};
string s2 {"Frank"};

string *p1 {&s1};
string *p2 {&s2};
string *p3 {&s1};

cout << *p1 == *p2 << endl; // true
cout << *p1 == *p3 << endl; // true

cout << p1 == p2 << endl; // false
cout << p1 == p3 << endl; // true
